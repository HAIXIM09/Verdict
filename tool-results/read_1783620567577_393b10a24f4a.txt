    23→export interface BattleCase {
    24→  id: string;
    25→  sideA: string;
    26→  sideB: string;
    27→  category: string;
    28→  difficulty: "Bronze" | "Silver" | "Gold" | "Platinum";
    29→  difficultyColor: string;
    30→  teamA: (User | null)[];
    31→  teamB: (User | null)[];
    32→  spectators: User[];
    33→  maxSpectators: number;
    34→  status: "waiting" | "live" | "verdict" | "finished";
    35→  timer: number;
    36→  totalTimer: number;
    37→  chatMessages: ChatMessage[];
    38→  aiVerdict: AiVerdict | null;
    39→  auraStake: number;
    40→  viewers: number;
    41→}
    42→
    43→export interface ChatMessage {
    44→  id: string;
    45→  userId: string;
    46→  username: string;
    47→  text: string;
    48→  timestamp: number;
    49→  isAi: boolean;
    50→  type: "argument" | "ai_kick" | "ai_warning" | "spectator_reaction" | "system";
    51→  reaction?: string;
    52→}
    53→
    54→export interface AiVerdict {
    55→  winner: "A" | "B";
    56→  scoreA: number;
    57→  scoreB: number;
    58→  bestArgument: { userId: string; username: string; text: string };

    60→  summary: string;
    61→  coinsAwarded: number;
    62→  auraChange: { gained: number; lost: number };
    63→}
    64→
    65→export interface Category {
    66→  id: string;
    67→  name: string;
    68→  icon: string;
    69→  battleCount: number;
    70→  color: string;
    71→}
    72→
    73→export interface Quest {
    74→  id: string;
    75→  title: string;
    76→  description: string;
    77→  progress: number;
    78→  target: number;
    79→  reward: { coins: number; aura: number };
    80→  completed: boolean;
    81→  claimed: boolean;
    82→}

    86→  name: string;

    92→  equipped: boolean;
    93→  previewType: "border" | "text" | "icon" | "description";
